# faker_rust::movies::hobbit

// Any character from the book
hobbit::character() //=> "Gandalf the Grey"

hobbit::thorins_company() //=> "Thorin Oakenshield"

hobbit::quote() //=> "Never laugh at live dragons, Bilbo you fool!"

hobbit::location() //=> "The Shire"
