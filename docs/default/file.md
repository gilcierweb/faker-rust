# faker_rust::default::file

available since version 1.6.4.

```rust
file::extension() //=> "mp3"

file::mime_type() //=> "application/pdf"
file::mime_type() //=> "image/png"

file::file_name() //=> "path/to/something_random::jpg"
file::file_name() //=> "foo/bar/baz::zip"
file::file_name() //=> "foo/bar/baz::doc"
file::file_name() //=> "foo/bar\\baz::mp3"

file::dir() //=> "et_error/sint_voluptas/quas_veritatis"
file::dir() //=> "ea-suscipit/ut-deleniti"
file::dir() //=> "est_porro/fugit_eveniet/incidunt-autem"
file::dir() //=> "aut-ullam\\quia_quisquam\\ut-eos"
```
